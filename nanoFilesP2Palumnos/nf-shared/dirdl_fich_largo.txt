package org.pds.negocio.controladores;

import java.util.List;

import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;

import org.pds.adapters.repository.TableroRepository;
import org.pds.adapters.repository.exceptions.TableroExistenteException;
import org.pds.adapters.repository.exceptions.TableroNoEncontradoException;
import org.pds.adapters.repository.exceptions.TableroPersistenciaException;
import org.pds.negocio.modelo.Checklist;
import org.pds.negocio.modelo.CorreoUsuario;
import org.pds.negocio.modelo.Descripcion;
import org.pds.negocio.modelo.Item;
import org.pds.negocio.modelo.ItemId;
import org.pds.negocio.modelo.ListaTareasId;
import org.pds.negocio.modelo.Nombre;
import org.pds.negocio.modelo.Tablero;
import org.pds.negocio.modelo.TarjetaId;
import org.pds.negocio.modelo.Tarjeta;

public class ControladorTableros {

    private static final Logger log = LogManager.getLogger();
    private TableroRepository repositorioTableros;

    public ControladorTableros(TableroRepository repositorio) {
        this.repositorioTableros = repositorio;
    }

    public List<Tablero> getTableros() {
        return repositorioTableros.getTableros();
    }

    public Tablero getTableroPorId(String tableroId) throws TableroNoEncontradoException {
        return repositorioTableros.findById(tableroId);
    }

    public Tablero crearTablero(String nombre, String emailPropietario) {
        Tablero tablero = new Tablero(new Nombre(nombre), new CorreoUsuario(emailPropietario));
        try {
            repositorioTableros.addTablero(tablero);
            log.info("Tablero creado: {}", nombre);
        } catch (TableroExistenteException | TableroPersistenciaException e) {
            log.error("Error creando tablero: {}", nombre, e);
        }
        return tablero;
    }

    public void eliminarTablero(Tablero tablero) {
        try {
            repositorioTableros.removeTablero(tablero);
        } catch (TableroPersistenciaException e) {
            log.error("Error eliminando tablero", e);
        }
    }

    public void bloquearTablero(Tablero tablero) {
        tablero.bloquearTablero();
        persistir(tablero);
    }
    public void desBloquearTablero(Tablero tablero) {
        tablero.desBloquearTablero();
        persistir(tablero);
    }

    public void agregarListaATablero(Tablero tablero, String nombreLista) {
        tablero.agregarLista(nombreLista);
        persistir(tablero);
    }

    public void eliminarListaDeTablero(Tablero tablero, ListaTareasId listaId) {
        tablero.eliminarLista(listaId);
        persistir(tablero);
    }

    public Tarjeta agregarTareaALista(Tablero tablero, ListaTareasId listaId,
                                       String nombre, String descripcion) {
        Tarjeta t = tablero.agregarTareaALista(listaId, nombre, descripcion);
        persistir(tablero);
        return t;
    }

    public Tarjeta agregarChecklistALista(Tablero tablero, ListaTareasId listaId, String nombre) {
        Tarjeta t = tablero.agregarCheckListALista(listaId, nombre);
        persistir(tablero);
        return t;
    }
    
    public void asignarEtiquetaATarjeta(Tablero tablero, String nombreEtiqueta, TarjetaId tarjetaId) {
        tablero.getListasDeEtiquetas().stream()
                .filter(e -> e.getNombre().valor().equals(nombreEtiqueta))
                .findFirst()
                .ifPresent(e -> {
                    tablero.agregarEtiquetaATarjeta(e.getId(), tarjetaId);
                    persistir(tablero);
                });
    }

    public void agregarItemAChecklist(Tablero tablero, ListaTareasId listaId,
                                       TarjetaId checklistId, String nombreItem) {
        tablero.getListasDeTareas().stream()
                .filter(l -> l.getId().equals(listaId))
                .findFirst()
                .flatMap(l -> l.buscarTarjeta(checklistId))
                .filter(t -> t instanceof Checklist)
                .map(t -> (Checklist) t)
                .ifPresent(c -> {
                    c.agregarItem(new Item(new Nombre(nombreItem)));
                    persistir(tablero);
                });
    }

    public void marcarItemChecklist(Tablero tablero, ListaTareasId listaId,
                                     TarjetaId checklistId, int indiceItem, boolean completado) {
        tablero.getListasDeTareas().stream()
                .filter(l -> l.getId().equals(listaId))
                .findFirst()
                .flatMap(l -> l.buscarTarjeta(checklistId))
                .filter(t -> t instanceof Checklist)
                .map(t -> (Checklist) t)
                .ifPresent(c -> {
                    List<Item> items = c.getItems();
                    if (indiceItem >= 0 && indiceItem < items.size()) {
                        if (completado) items.get(indiceItem).marcarComoCompletado();
                        else           items.get(indiceItem).desmarcar();
                        persistir(tablero);
                    }
                });
    }

    public void completarTarjeta(Tablero tablero, TarjetaId tarjetaId) {
        tablero.moverTarjetaACompletadas(tarjetaId);
        persistir(tablero);
    }

    public void eliminarTarjetaDeLista(Tablero tablero, ListaTareasId listaId, TarjetaId tarjetaId) {
        tablero.eliminarTarjetaDeLista(listaId, tarjetaId);
        persistir(tablero);
    }

    public void editarLista(Tablero tablero, ListaTareasId listaId, String nombre) {
    	 tablero.editarLista(listaId, nombre);
    	 persistir(tablero);
    }
    
    public void editarTarjeta(Tablero tablero, ListaTareasId listaId, String nombre) {
    	tablero.cambiarNombreTarjeta(listaId, nombre);
    	 persistir(tablero);
    }
    
    public void editarDescripcion(Tablero tablero, ListaTareasId listaId, TarjetaId tarjetaId, String descripcion) {
    	tablero.editarDescripcionTarea(listaId, tarjetaId, descripcion);
   	 	persistir(tablero);
   }
    
    public void eliminarItemChecklist(Tablero tablero, ListaTareasId listaId, TarjetaId tarjetaId, ItemId item) {
      	 tablero.eliminarElementoCheckList(listaId, tarjetaId, item);
      	 persistir(tablero);
    }


    private void persistir(Tablero tablero) {
        try {
            repositorioTableros.updateTablero(tablero);
        } catch (TableroPersistenciaException e) {
            log.error("Error persistiendo tablero", e);
        }
    }

    public void setTableroRepository(TableroRepository repositorio) {
        this.repositorioTableros = repositorio;
    }
}
